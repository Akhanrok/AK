cd / //перейти до кореневого каталогу
cd .. //перейти до батьківського каталогу
cd - //перейти до попереднього каталогу
ls //переглянути вміст поточного каталогу
ls -l //переглянути детальну інформацію про всі файли/підкаталоги (права (1 колонка), власника (3 колонка), групу власника (4 колонка) файлу (8 колонка))
man ls //відкрити довідку про команду (Q - вийти з довідки)
sudo su - //переключитися на адміністратора
chmod ug+w 1.txt //надати право власнику та групі на редагування
chmod 777 2.txt //надати всі права для всіх користувачів
echo 'Hello!' //вивести текст
cp 1.txt user //копіювати файл 1.txt у каталог user

tail 3.txt //переглянути останні 10 рядків файлу
tail -n5 3.txt //переглянути останні 5 рядків файлу
